METHOD wordCounter(paragraph)
    cleanText = remove punctuation, convert to lowercase
    words = split cleanText by spaces
    map = empty Map<String, Integer>

    FOR each word in words
        IF map contains word
            map[word]++
        ELSE
            map[word] = 1

    sort map by frequency (descending)

input = "The news today is about the rise in technology trends."
wordCounter(input)
