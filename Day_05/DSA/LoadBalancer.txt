class LoadBalancer {
    Queue<String> servers;

    void addServer(String s) { servers.add(s); }
    void removeServer(String s) { servers.remove(s); }

    String assignRequest(String request) {
        String server = servers.poll();   // take from front
        servers.add(server);              // push to back
        print(request + " â†’ " + server);
        return server;
    }
}

// Usage
main() {
    LoadBalancer lb = new LoadBalancer();
    lb.addServer("S1");
    lb.addServer("S2");
    lb.addServer("S3");

    lb.assignRequest("R1"); // S1
    lb.assignRequest("R2"); // S2
    lb.assignRequest("R3"); // S3

    lb.addServer("S4");
    lb.assignRequest("R4"); // S1 again
    lb.removeServer("S2");  // server fails
}
